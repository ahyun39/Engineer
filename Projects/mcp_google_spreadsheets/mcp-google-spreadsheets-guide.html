🏗 전체 구조

Claude가 자연어로 지시를 내리면 MCP 서버가 중간에서 Google Apps Script(GAS)로 HTTP 요청을 보냅니다. GAS는 스프레드시트를 직접 읽고 쓴 뒤 결과를 반환합니다.

📋 구현 단계

1

Google Apps Script 웹 앱 만들기

스프레드시트 → 확장 프로그램 → Apps Script

GAS가 HTTP 엔드포인트 역할을 합니다. MCP 서버가 JSON을 POST하면 doPost()가 시트를 읽거나 씁니다. 아래 순서대로 진행하세요.

  1. 1스프레드시트를 열고 상단 메뉴에서 확장 프로그램 → Apps Script를 클릭해 편집기로 들어갑니다.
  2. 2편집기에 기본으로 작성된 코드를 전부 지우고 아래 코드를 복사해 붙여넣으세요.
  3. 3편집기 상단에 "제목 없는 프로젝트"라고 되어있는 이름을 원하는 이름으로 수정하세요. (예: Claude MCP Server)
  4. 4Ctrl+S 또는 저장 버튼을 눌러 코드를 저장합니다.
  5. 5오른쪽 상단 배포 → 새 배포를 클릭합니다. 유형 선택 화면에서 톱니바퀴(도구) 버튼을 눌러 웹 앱을 선택하세요. 다음 사용자로 실행: 나, 액세스 권한이 있는 사용자: 모든 사용자로 설정한 뒤 배포를 누릅니다.
  6. 6액세스 승인 버튼이 나타나면 클릭합니다. Google 계정 선택 팝업이 열리면 계정을 선택하고, 경고 화면에서 Advanced를 클릭 → Go to [프로젝트명] (unsafe) 클릭 → Continue를 눌러 권한을 승인합니다.
  7. 7배포가 완료되면 나타나는 팝업에서 웹 앱 URL을 복사합니다. 이 URL이 MCP 서버의 엔드포인트입니다.
  8. 8나중에 URL을 다시 확인하려면 배포 → 배포 관리를 눌러 Claude MCP Server 항목을 열면 웹 앱 URL을 다시 복사할 수 있습니다.
Code.gs (Google Apps Script) 
/**
 * Claude MCP Server for Google Apps Script
 * 클로드와 구글 워크스페이스를 연결하는 MCP 서버
 */
// MCP 서버 설정
const MCP_CONFIG = {
  name: "google-workspace-mcp",
  version: "1.0.0",
  description: "Google Workspace integration for Claude via MCP",
};
// 도구 목록 정의
const TOOLS = {
  READ_SHEET: {
    name: "read_google_sheet",
    description: "Google Sheets에서 데이터를 읽어옵니다",
    inputSchema: {
      type: "object",
      properties: {
        spreadsheetId: {
          type: "string",
          description: "스프레드시트 ID",
        },
        range: {
          type: "string",
          description: "읽을 범위 (예: A1:C10)",
          default: "A:Z",
        },
      },
      required: ["spreadsheetId"],
    },
  },
  WRITE_SHEET: {
    name: "write_google_sheet",
    description: "Google Sheets에 데이터를 씁니다",
    inputSchema: {
      type: "object",
      properties: {
        spreadsheetId: {
          type: "string",
          description: "스프레드시트 ID",
        },
        range: {
          type: "string",
          description: "쓸 범위 (예: A1:C10)",
        },
        values: {
          type: "array",
          description: "쓸 데이터 (2차원 배열)",
        },
      },
      required: ["spreadsheetId", "range", "values"],
    },
  },
};
// MCP 서버로부터 POST 요청을 받는 진입점
// Claude → Node.js MCP 서버 → 여기(GAS 웹 앱) 순서로 호출됩니다
function doPost(e) {
  try {
    const request = JSON.parse(e.postData.contents); // 요청 본문을 JSON으로 파싱
    console.log("MCP Request:", request);
    // 비밀키 검증 — 스크립트 속성에 MCP_SECRET이 설정된 경우에만 검사합니다
    const secret = PropertiesService.getScriptProperties().getProperty("MCP_SECRET");
    if (secret && request.secret !== secret) return createErrorResponse("Unauthorized");

    if (request.method === "tools/call") {
      return createResponse(handleToolCall(request.params));
    }
    return createErrorResponse("Unknown method: " + request.method);
  } catch (error) {
    console.error("MCP Error:", error);
    return createErrorResponse("Server error: " + error.message);
  }
}

/**
 * 도구 목록 반환
 */
function handleToolsList() {
  return {
    tools: Object.values(TOOLS),
  };
}

// 요청된 도구 이름에 따라 읽기/쓰기 함수로 분기합니다
function handleToolCall(params) {
  const { name, arguments: args } = params;
  try {
    if (name === "read_google_sheet") return readGoogleSheet(args);
    if (name === "write_google_sheet") return writeGoogleSheet(args);
    throw new Error("Unknown tool: " + name);
  } catch (error) {
    return { isError: true, content: [{ type: "text", text: `도구 실행 오류: ${error.message}` }] };
  }
}

// 지정한 범위의 데이터를 읽어 2차원 배열로 반환합니다
// 빈 행은 자동으로 제거됩니다
function readGoogleSheet(args) {
  const { spreadsheetId, range = "A:Z" } = args;
  // spreadsheetId 화이트리스트 — 스크립트 속성 ALLOWED_IDS에 등록된 ID만 허용합니다
  const allowed = PropertiesService.getScriptProperties().getProperty("ALLOWED_IDS");
  if (allowed && !allowed.split(",").map(s => s.trim()).includes(spreadsheetId)) throw new Error("허가되지 않은 spreadsheetId");
  const sheet = SpreadsheetApp.openById(spreadsheetId).getActiveSheet();
  const data = sheet.getRange(range).getValues();
  const filteredData = data.filter((row) => row.some((cell) => cell !== "")); // 빈 행 제거
  return {
    content: [{ type: "text", text: `스프레드시트 데이터 (${filteredData.length}행):\n${JSON.stringify(filteredData, null, 2)}` }],
  };
}

// 지정한 범위에 2차원 배열 데이터를 씁니다
// range 크기와 values 크기가 일치해야 합니다 (예: A1:C3 → 3행×3열)
function writeGoogleSheet(args) {
  const { spreadsheetId, range, values } = args;
  if (!values.length) throw new Error("values가 비어있습니다");
  // spreadsheetId 화이트리스트 — 스크립트 속성 ALLOWED_IDS에 등록된 ID만 허용합니다
  const allowed = PropertiesService.getScriptProperties().getProperty("ALLOWED_IDS");
  if (allowed && !allowed.split(",").map(s => s.trim()).includes(spreadsheetId)) throw new Error("허가되지 않은 spreadsheetId");
  const sheet = SpreadsheetApp.openById(spreadsheetId).getActiveSheet();
  const target = sheet.getRange(range);
  // range와 values 크기가 다르면 GAS에서 오류가 나므로 미리 검증합니다
  if (target.getNumRows() !== values.length || target.getNumColumns() !== values[0].length) {
    throw new Error(
      `range ${range} (${target.getNumRows()}×${target.getNumColumns()})와 values (${values.length}×${values[0].length}) 크기가 다릅니다`
    );
  }
  target.setValues(values);
  return {
    content: [{ type: "text", text: `${values.length}행의 데이터를 ${range}에 저장했습니다.` }],
  };
}

// 성공 응답: JSON 문자열을 ContentService로 감싸서 반환합니다
function createResponse(data) {
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

// 오류 응답: isError 플래그를 true로 설정해 MCP 서버가 에러를 인식하게 합니다
function createErrorResponse(message) {
  return ContentService.createTextOutput(
    JSON.stringify({ isError: true, content: [{ type: "text", text: message }] })
  ).setMimeType(ContentService.MimeType.JSON);
}

/**
 * 테스트 함수
 */
function testMCPServer() {
  console.log("MCP 서버 테스트 시작");
  // 도구 목록 테스트
  const toolsList = handleToolsList();
  console.log("도구 목록:", toolsList);
  console.log("MCP 서버가 정상적으로 설정되었습니다!");
}

// GAS 편집기에서 직접 실행해 읽기 동작을 테스트하는 함수입니다
function testReadSheet() {
  const spreadsheetId = "YOUR_SPREADSHEET_ID"; // 스프레드시트 url에서 YOUR_SPREADSHEET_ID 부분에 해당하는 ID를 복사하세요. (https://docs.google.com/spreadsheets/d/YOUR_SPREADSHEET_ID/edit?gid=0#gid=0)
  try {

    console.error("스프레드시트 읽기 실패:", error);
  }
}

MCP 서버 코드

src/server.js
src/server.js 
// MCP SDK: Claude와 stdio로 통신하는 서버 구현체
import { Server } from "@modelcontextprotocol/sdk/server/index.js";
import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js";
import { CallToolRequestSchema, ListToolsRequestSchema } from "@modelcontextprotocol/sdk/types.js";
import { readFileSync } from "fs";

// 설정 파일에서 기본값을 읽어옵니다
const config = JSON.parse(readFileSync(new URL("../config/mcp-config.json", import.meta.url), "utf8"));
// GAS 웹 앱 URL은 환경변수 우선 — 소스에 URL이 노출되지 않도록 env로 주입합니다
const webAppUrl = process.env.GAS_WEB_APP_URL || config.webAppUrl;
if (!webAppUrl) throw new Error("GAS_WEB_APP_URL 환경변수 또는 config.webAppUrl 설정 필요");

// MCP 서버 인스턴스 생성
const server = new Server(
  { name: "google-apps-script-mcp", version: "1.0.0" },
  { capabilities: { tools: {} } } // 이 서버가 tools 기능을 제공한다고 선언
);

server.onerror = (error) => console.error("[MCP Error]", error);
process.on("SIGINT", async () => { await server.close(); process.exit(0); }); // Ctrl+C 종료 처리

// ① Claude가 "어떤 도구를 쓸 수 있어?" 라고 물을 때 응답합니다
// Claude는 이 목록을 보고 대화 중 적절한 도구를 자동으로 선택합니다
server.setRequestHandler(ListToolsRequestSchema, async () => ({
  tools: [
    {
      name: "read_google_sheet",
      description: "Google Sheets에서 데이터를 읽어옵니다",
      inputSchema: {
        type: "object",
        properties: {
          spreadsheetId: { type: "string", description: "스프레드시트 ID" },
          range: { type: "string", description: "읽을 범위 (예: A1:C10)", default: "A:Z" },
        },
        required: ["spreadsheetId"],
      },
    },
    {
      name: "write_google_sheet",
      description: "Google Sheets에 데이터를 씁니다",
      inputSchema: {
        type: "object",
        properties: {
          spreadsheetId: { type: "string", description: "스프레드시트 ID" },
          range: { type: "string", description: "쓸 범위 (예: A1:C10)" },
          values: { type: "array", description: "쓸 데이터 (2차원 배열)" },
        },
        required: ["spreadsheetId", "range", "values"],
      },
    },
  ],
}));

// ② Claude가 실제로 도구를 실행할 때 호출됩니다
// Claude의 요청을 그대로 GAS 웹 앱에 전달하고, 응답을 Claude에게 돌려줍니다
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  const { name, arguments: args } = request.params;
  try {
    if (!["read_google_sheet", "write_google_sheet"].includes(name)) {
      throw new Error(`알 수 없는 도구: ${name}`);
    }
    // GAS 웹 앱에 HTTP POST로 요청 전달 (30초 타임아웃)
    const response = await fetch(webAppUrl, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ method: "tools/call", params: { name, arguments: args }, ...(process.env.MCP_SECRET && { secret: process.env.MCP_SECRET }) }),
      signal: AbortSignal.timeout(30000),
    });
    const data = await response.json();
    return { content: [{ type: "text", text: data.content?.[0]?.text || "완료" }] };
  } catch (error) {
    return { content: [{ type: "text", text: `오류 발생: ${error.message}` }], isError: true };
  }
});

// stdio 트랜스포트로 Claude Desktop과 연결 시작
const transport = new StdioServerTransport();
await server.connect(transport);
console.error("Google Apps Script MCP 서버가 시작되었습니다.");
// URL의 deploy ID는 로그에 마스킹 — 보안상 민감한 정보가 로그에 남지 않도록 합니다
console.error(`웹 앱 URL: ${webAppUrl.replace(/\/s\/[^/]+\//, "/s/[REDACTED]/")}`);
